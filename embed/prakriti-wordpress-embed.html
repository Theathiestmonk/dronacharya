2b) Mobile keyboard — lift the chat block above the on-screen keyboard (embed only)

On prakritischool.ai/chat (no iframe), the app moves the input bar using the browser’s Visual Viewport. In WordPress, the chat runs in a fixed-height iframe: the app does not add extra bottom margin inside the iframe (doing that and lifting the wrapper caused a huge gap between the field and the keyboard). Add this script on the same page as the embed so the parent page lifts .prakriti-chat-embed using the top window’s visualViewport when the keyboard opens. Paste into Divi → Theme Options → Integration → Body, a Code module, or your theme footer.

<script>
(function () {
  var sel = '.prakriti-chat-embed';
  function overlapPx() {
    if (!window.visualViewport) return 0;
    var vv = window.visualViewport;
    return Math.max(0, window.innerHeight - vv.offsetTop - vv.height);
  }
  function lift() {
    var px = overlapPx();
    document.querySelectorAll(sel).forEach(function (wrap) {
      wrap.style.transform = px > 0 ? 'translateY(-' + px + 'px)' : '';
      wrap.style.transition = 'transform 0.15s ease-out';
      wrap.style.willChange = px > 0 ? 'transform' : 'auto';
    });
  }
  if (window.visualViewport) {
    window.visualViewport.addEventListener('resize', lift);
    window.visualViewport.addEventListener('scroll', lift);
  }
  window.addEventListener('resize', lift);
})();
</script>

Deploy the latest chat app so the iframe does not double-lift, then hard-refresh the school page. If the bar still clears the keyboard too much or too little, nudge mobile .prakriti-chat-embed height in your CSS. Change sel if your wrapper class differs.
